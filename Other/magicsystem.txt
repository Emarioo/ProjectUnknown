Scroll:
	Static Variables: Caster

"The Force Spell"
var target
var dist
upon LeftDown
	target = Target Caster
	dist = Length target.Position - Caster.Position
end
upon LeftUp
	target = none
end
upon MoveMouse
	if target
		target.Position = Caster.Position + Caster.Look * dist
	end
end

"LEVITATE"
upon LeftDown
	target = Target Caster
	target.Gravity = false
	target.Velocity.y = 0.1
end
upon LeftUp
	target = Target Caster
	target.Gravity = true
	target.Velocity.y = 0
end

"FIRE!"
upon LeftDown
	target = Raycast Caster 20
	targets = Targets target.position 5
	targets.Fire = true
end
